'''6. Factorial Calculator
 Write a program that prompts the user to enter a non-negative integer.
o Calculate the factorial of the number (n!) using a loop.
o Display the result.'''

def Factorial():
    n=int(input("enter a non-negative Integer : "))
    i=n
    while(i>=2):
        
        n=n*(i-1)
        i=i-1
    return n
print(Factorial())

'''7. Palindrome Checker
 Write a program that prompts the user to enter a word or phrase.
o Check if the entered text is a palindrome (reads the same forwards and backwards).
o Print "The text is a palindrome" or "The text is not a palindrome" based on the result.'''

def main():
    user_input = input("Enter a word or phrase: ")
    text = ''.join(user_input.split()).lower()
    if text == text[::-1]:
        print("The text is a palindrome")
    else:
        print("The text is not a palindrome")
main()

'''8. List Sum Calculator
 Write a program that initializes a list of numbers.
o Calculate and print the sum of all numbers in the list.
o Prompt the user to enter a number to add to the list.
o Update the list and print the new sum'''


def Sum_Calc():
    numbers = [1, 2, 3, 4, 5]
    cur_sum = sum(numbers)
    print(f"The sum of the initial list is: {cur_sum}")
 
    try:
        new_number = float(input("Enter a number to add to the list: "))
        numbers.append(new_number)
        new_sum = sum(numbers)
        print(f"The new sum after adding {new_number} is: {new_sum}")
    except ValueError:
        print("Invalid input. Please enter a numeric value.")
    return 0  
Sum_Calc()
